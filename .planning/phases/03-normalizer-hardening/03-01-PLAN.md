    ```typescript
    function stripMarkdown(text: string): string {
      // Order matters: more specific patterns before general ones (per RESEARCH.md Pitfall 4)

      // 1. Code block fences -- line-anchored for ReDoS safety (0.09ms vs 862ms with [\s\S]*?)
      text = text.replace(/^```[^\n]*$/gm, '');

      // 2. Inline code backticks
      text = text.replace(/`([^`]+)`/g, '$1');

      // 3. Images ![alt](url) -- before links to avoid overlap
      text = text.replace(/!\[([^\]]*)\]\([^)]*\)/g, '$1');

      // 4. Links [text](url)
      text = text.replace(/\[([^\]]*)\]\([^)]*\)/g, '$1');

      // 5. Bold/italic (** and __ before * and _ to handle correctly)
      text = text.replace(/(\*\*|__)(.*?)\1/g, '$2');
      text = text.replace(/(\*|_)(.*?)\1/g, '$2');

      // 6. Headers (# to ######)
      text = text.replace(/^#{1,6}\s+/gm, '');

      return text;
    }
    ```
